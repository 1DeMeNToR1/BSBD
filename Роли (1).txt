--РОЛЬ ДЛЯ СОТРУДНИКОВ
CREATE ROLE Сотрудник WITH
	LOGIN
	NOSUPERUSER
	INHERIT
	NOCREATEDB
	CREATEROLE
	NOREPLICATION;

	LOGIN

--РОЛЬ ДЛЯ КЛИЕНТСКОГО АДМИНИСТРАТОРА
