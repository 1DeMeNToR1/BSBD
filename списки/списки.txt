	CREATE OR REPLACE FUNCTION Маскировка_СТС(Номер_СТС TEXT) RETURNS TEXT AS $$
	BEGIN
    RETURN CONCAT(SUBSTRING(Номер_СТС FROM 1 FOR 4), '****', SUBSTRING(Номер_СТС FROM 9));
	END;
	$$ LANGUAGE plpgsql;

Шифровка:
	1)Паспорт сотрудника
	CREATE EXTENSION IF NOT EXISTS pgcrypto;
	
	UPDATE "Сотрудники"
	SET "Паспорт" = pgp_sym_encrypt("Паспорт", 'ключ', 'compress-algo=1, cipher-algo=aes256');
	
	SELECT pgp_sym_decrypt("Паспорт", 'ключ', 'compress-algo=1, cipher-algo=aes256') AS Расшифрованный_Паспорт
	FROM "Сотрудники";
